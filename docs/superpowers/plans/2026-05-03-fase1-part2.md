# quantconnect/signal_data.py
"""
QuantConnect Lean PythonData custom feed for LLM signals.
Lives in the quantconnect/ directory; QC imports it at algorithm init.
Reads live signals from http://localhost:8000/api/signals/{symbol}
Reads historical signals from http://localhost:8000/api/signals/history?symbol=...&date=...
"""
import json
from datetime import datetime
from AlgorithmImports import *   # noqa — provided by QC runtime

SIGNAL_API_BASE = "http://localhost:8000"
SIGNAL_MAX_AGE_MIN = 30  # ignore signal if older than 2× worker interval


class LLMSignalData(PythonData):

    def GetSource(self, config, date, isLive):
        symbol = config.Symbol.Value
        if isLive:
            url = f"{SIGNAL_API_BASE}/api/signals/{symbol}"
        else:
            url = f"{SIGNAL_API_BASE}/api/signals/history?symbol={symbol}&date={date:%Y-%m-%d}"
        return SubscriptionDataSource(url, SubscriptionTransportMedium.Rest)

    def Reader(self, config, line, date, isLive):
        if not line.strip():
            return None
        try:
            data = json.loads(line)
        except (ValueError, json.JSONDecodeError):
            return None

        signal = LLMSignalData()
        signal.Symbol = config.Symbol
        signal.Time = datetime.fromisoformat(data["generated_at"].replace("Z", "+00:00"))
        signal.Value = float(data.get("score", 0.0))
        signal["sentiment_score"]    = float(data.get("score", 0.0))
        signal["regime_multiplier"]  = float(data.get("regime_multiplier", 1.0))
        signal["confidence"]         = float(data.get("confidence", 0.0))
        signal["fallback_used"]      = bool(data.get("fallback_used", False))
        signal["generated_at"]       = data.get("generated_at", "")
        return signal

    @staticmethod
    def is_fresh(signal: "LLMSignalData", algorithm_time: datetime) -> bool:
        try:
            gen_at = datetime.fromisoformat(signal["generated_at"].replace("Z", "+00:00"))
            age_min = (algorithm_time - gen_at).total_seconds() / 60
            return age_min <= SIGNAL_MAX_AGE_MIN
        except (KeyError, ValueError, TypeError):
            return False
```

# quantconnect/intraday_strategy.py
"""
Intraday 1h strategy using pre-computed LLM sentiment signals.
Entry: sentiment_score > ENTRY_THRESHOLD and EMA momentum confirmed.
Risk: 2% stop-loss, max 10% position size, rate limit 10 orders/min.
Kill-switch: checked via Redis at every OnData() call.
"""
import json
import redis
from AlgorithmImports import *   # noqa
from quantconnect.signal_data import LLMSignalData

ENTRY_THRESHOLD = 0.3
MAX_POSITION_PCT = 0.10
STOP_LOSS_PCT = 0.02
MAX_ORDERS_PER_MIN = 10
REDIS_URL = "redis://localhost:6379/0"

_SYMBOLS = ["AAPL", "MSFT", "SPY", "QQQ"]  # override in config


class LLMIntradayStrategy(QCAlgorithm):

    def Initialize(self):
        self.SetStartDate(2022, 1, 1)
        self.SetEndDate(2024, 12, 31)
        self.SetCash(100_000)
        self.SetBrokerageModel(BrokerageName.InteractiveBrokersBrokerage)

        self._redis = redis.from_url(REDIS_URL)
        self._order_times: list[datetime] = []
        self._stop_prices: dict[str, float] = {}

        for sym in _SYMBOLS:
            equity = self.AddEquity(sym, Resolution.Hour)
            self.AddData(LLMSignalData, sym, Resolution.Minute)
            equity.SetDataNormalizationMode(DataNormalizationMode.Adjusted)

        # EMA for momentum confirmation
        self._emas = {s: self.EMA(s, 20, Resolution.Hour) for s in _SYMBOLS}

    def OnData(self, data: Slice):
        # Check kill-switch every tick
        if self._redis.get("killswitch_active"):
            self.Liquidate("Kill-switch active")
            return

        for sym in _SYMBOLS:
            if not data.ContainsKey(sym):
                continue

            signal_key = f"{sym}_llmsignaldata"
            if not data.ContainsKey(signal_key):
                continue

            signal = data[signal_key]
            if not LLMSignalData.is_fresh(signal, self.Time):
                continue

            score = signal["sentiment_score"]
            regime_mult = signal["regime_multiplier"]
            ema = self._emas[sym]

            if not ema.IsReady:
                continue

            price = self.Securities[sym].Price
            holding = self.Portfolio[sym]

            # Entry
            if not holding.Invested and score > ENTRY_THRESHOLD:
                momentum_ok = price > ema.Current.Value
                if momentum_ok and self._can_place_order():
                    size = self.Portfolio.TotalPortfolioValue * MAX_POSITION_PCT * regime_mult / price
                    self.MarketOrder(sym, int(size))
                    self._stop_prices[sym] = price * (1 - STOP_LOSS_PCT)
                    self._order_times.append(self.Time)

            # Stop-loss
            if holding.Invested and sym in self._stop_prices:
                if price < self._stop_prices[sym]:
                    self.Liquidate(sym)
                    del self._stop_prices[sym]

    def _can_place_order(self) -> bool:
        now = self.Time
        recent = [t for t in self._order_times if (now - t).total_seconds() < 60]
        self._order_times = recent
        return len(recent) < MAX_ORDERS_PER_MIN
```

# src/performance/ic.py
"""
Composite IC (B4): 0.5 × Spearman + 0.3 × weighted_hit_rate + 0.2 × (1 − Brier)
ICIR via Newey-West HAC standard errors to correct for autocorrelation.
"""
import numpy as np
from scipy.stats import spearmanr


def composite_ic(
    scores: list[float],
    returns: list[float],
    confidences: list[float],
) -> float:
    """Composite IC = 0.5×Spearman + 0.3×weighted_hit_rate + 0.2×(1−Brier)."""
    if len(scores) < 5:
        return 0.0

    s = np.array(scores)
    r = np.array(returns)
    c = np.array(confidences)

    spearman = float(spearmanr(s, r).statistic)
    if np.isnan(spearman):
        spearman = 0.0

    whr = weighted_hit_rate(scores, returns, confidences)
    brier = float(np.mean((c - (r > 0).astype(float)) ** 2))

    return 0.5 * spearman + 0.3 * whr + 0.2 * (1.0 - brier)


def weighted_hit_rate(
    scores: list[float],
    returns: list[float],
    confidences: list[float],
) -> float:
    """Confidence-weighted fraction of signals with correct sign."""
    s = np.array(scores)
    r = np.array(returns)
    c = np.array(confidences)
    correct = (np.sign(s) == np.sign(r)).astype(float)
    total_conf = c.sum()
    if total_conf == 0:
        return 0.0
    return float((correct * c).sum() / total_conf)


def compute_ic_series(
    scores: list[float],
    returns: list[float],
    confidences: list[float],
    window: int = 30,
    step: int = 5,
) -> list[float]:
    """Sliding-window composite IC series for ICIR computation."""
    series = []
    for i in range(0, len(scores) - window + 1, step):
        s_w = scores[i: i + window]
        r_w = returns[i: i + window]
        c_w = confidences[i: i + window]
        series.append(composite_ic(s_w, r_w, c_w))
    return series


def icir_newey_west(ic_series: list[float], lags: int = 4) -> float:
    """ICIR = IC_mean / NW_std. Returns 0 if fewer than 10 observations."""
    if len(ic_series) < 10:
        return 0.0

    x = np.array(ic_series)
    n = len(x)
    mu = x.mean()
    residuals = x - mu

    # Newey-West HAC variance
    gamma_0 = float(np.dot(residuals, residuals) / n)
    nw_var = gamma_0
    for lag in range(1, lags + 1):
        weight = 1.0 - lag / (lags + 1)
        gamma_k = float(np.dot(residuals[lag:], residuals[:-lag]) / n)
        nw_var += 2.0 * weight * gamma_k

    nw_std = max(float(np.sqrt(abs(nw_var))), 1e-8)
    return float(mu / nw_std)


def exclude_gap_days(
    scores: list[float],
    returns: list[float],
    confidences: list[float],
    dates: list[str],
    gap_threshold: float = 0.05,
) -> tuple[list[float], list[float], list[float]]:
    """Exclude days where |return| > gap_threshold (overnight gap event)."""
    filtered = [
        (s, r, c) for s, r, c, d in zip(scores, returns, confidences, dates)
        if abs(r) <= gap_threshold
    ]
    if not filtered:
        return [], [], []
    s_f, r_f, c_f = zip(*filtered)
    return list(s_f), list(r_f), list(c_f)
```

# src/performance/weights.py
"""
Leave-one-out IC (B3) corrects chicken-and-egg bias in ensemble weighting.
Smoothing α=0.25 (PW-Q1): new = 75% old + 25% target.
Bounds: floor 10%, cap 70%, max delta 10% per update (PW-Q2).
In Fase 1: compute_new_weights is OBSERVATIONAL — call should_auto_apply()
to check whether to apply. Performance Worker in Fase 1 only reports.
"""
import numpy as np
from scipy.stats import spearmanr
from src.performance.ic import composite_ic

_WEIGHT_FLOOR = 0.10
_WEIGHT_CAP   = 0.70
_MAX_DELTA    = 0.10
_MIN_SAMPLES  = 300
_ICIR_MIN_AUTO_APPLY = 0.10


def compute_purified_icir(
    model_signals: dict[str, list[float]],
    forward_returns: list[float],
    confidences: list[float],
    current_weights: dict[str, float],
    window: int = 30,
    step: int = 5,
) -> dict[str, float]:
    """Leave-one-out ICIR: IC of model X is measured against ensemble excluding X."""
    purified: dict[str, float] = {}
    models = list(model_signals.keys())

    for target in models:
        others = [m for m in models if m != target]
        n = len(forward_returns)

        # Ensemble scores excluding target
        loo_scores = []
        other_weight_sum = sum(current_weights[m] for m in others) or 1.0
        for i in range(n):
            s = sum(
                model_signals[m][i] * current_weights[m] / other_weight_sum
                for m in others
            )
            loo_scores.append(s)

        # Sliding-window IC series
        ic_series = []
        for w in range(0, n - window + 1, step):
            s_w = loo_scores[w: w + window]
            r_w = forward_returns[w: w + window]
            c_w = confidences[w: w + window]
            ic_series.append(composite_ic(s_w, r_w, c_w))

        if len(ic_series) < 3:
            purified[target] = 0.0
        else:
            mu = np.mean(ic_series)
            std = np.std(ic_series) + 1e-8
            purified[target] = float(mu / std)

    return purified


def compute_new_weights(
    purified_icir: dict[str, float],
    current_weights: dict[str, float],
    alpha: float = 0.25,
) -> dict[str, float]:
    """Smoothed target weights with floor/cap/delta guardrails."""
    # Map negative ICIR to 0 (don't reward bad models)
    raw = {m: max(0.0, icir) for m, icir in purified_icir.items()}
    total = sum(raw.values()) or 1.0
    target = {m: v / total for m, v in raw.items()}

    # Smoothing: 75% old + 25% new target
    blended = {m: (1.0 - alpha) * current_weights[m] + alpha * target[m] for m in target}

    # Floor / cap
    clipped = {m: max(_WEIGHT_FLOOR, min(_WEIGHT_CAP, w)) for m, w in blended.items()}

    # Max delta per update
    clipped = {
        m: max(current_weights[m] - _MAX_DELTA, min(current_weights[m] + _MAX_DELTA, w))
        for m, w in clipped.items()
    }

    # Re-normalize to sum to 1.0
    total_clipped = sum(clipped.values()) or 1.0
    return {m: w / total_clipped for m, w in clipped.items()}


def should_auto_apply(
    sample_count: int,
    icir_overall: float,
    freeze_reason: str,
) -> bool:
    """
    In Fase 1: always returns False (observational mode only).
    In Fase 2: auto-apply if sample_count >= 300, icir > 0.1, no circuit breaker.
    """
    if freeze_reason:
        return False
    if sample_count < _MIN_SAMPLES:
        return False
    if icir_overall < _ICIR_MIN_AUTO_APPLY:
        return False
    return True   # Fase 2 will flip this; Fase 1 workers should treat as False
```

# src/performance/drift.py
from dataclasses import dataclass
from enum import Enum
import numpy as np


class DriftLevel(Enum):
    STABLE = "stable"
    YELLOW = "yellow"   # only primary baseline (90gg) flags drift
    RED    = "red"      # both baselines flag drift


@dataclass
class MarketContext:
    vix: float
    vix_1d_change: float
    portfolio_drawdown: float
    consecutive_negative_ic_days: int
    portfolio_earnings_pct: float
    cross_asset_correlation: float


def compute_psi(baseline: np.ndarray, current: np.ndarray, bins: int = 10) -> float:
    """PSI = Σ (expected_i − actual_i) × ln(expected_i / actual_i)."""
    lo = min(baseline.min(), current.min())
    hi = max(baseline.max(), current.max())
    if lo == hi:
        return 0.0
    edges = np.linspace(lo, hi, bins + 1)
    exp = np.histogram(baseline, edges)[0] / len(baseline) + 1e-9
    act = np.histogram(current,  edges)[0] / len(current)  + 1e-9
    return float(np.sum((exp - act) * np.log(exp / act)))


def cusum_detect(
    recent_values: list[float],
    baseline_mean: float,
    baseline_std: float,
    threshold: float = 5.0,
) -> bool:
    """Cumulative sum test: returns True if CUSUM exceeds threshold."""
    cusum = 0.0
    for v in recent_values:
        cusum += (v - baseline_mean) / (baseline_std + 1e-8)
        if abs(cusum) >= threshold:
            return True
    return False


def check_drift(
    baseline_90d: np.ndarray,
    baseline_12m: np.ndarray,
    current: np.ndarray,
    psi_yellow: float = 0.10,
    psi_red: float    = 0.25,
    psi_12m_threshold: float = 0.10,
) -> DriftLevel:
    """
    YELLOW = PSI_90d > psi_yellow  (primary only)
    RED    = PSI_90d > psi_red AND PSI_12m > psi_12m_threshold
    """
    psi_90 = compute_psi(baseline_90d, current)
    psi_12 = compute_psi(baseline_12m, current)

    if psi_90 > psi_red and psi_12 > psi_12m_threshold:
        return DriftLevel.RED
    if psi_90 > psi_yellow:
        return DriftLevel.YELLOW
    return DriftLevel.STABLE


# --- Circuit breakers ---

def should_freeze_weight_update(ctx: MarketContext) -> tuple[bool, str]:
    """Returns (freeze: bool, reason: str). Hard breakers only."""
    if ctx.vix > 40.0 or ctx.vix_1d_change > 0.30:
        return True, "vix_spike"
    if ctx.portfolio_drawdown > 0.05:
        return True, "system_drawdown"
    if ctx.consecutive_negative_ic_days >= 5:
        return True, "ic_negative_run"
    return False, ""


def soft_warnings(ctx: MarketContext) -> list[str]:
    """Returns list of active soft warning names (appear in report, don't block)."""
    warnings = []
    if ctx.portfolio_earnings_pct > 0.50:
        warnings.append("earnings_concentration")
    if ctx.cross_asset_correlation > 0.90:
        warnings.append("cross_asset_corr")
    return warnings
```

# src/performance/postmortem.py
"""
Post-mortem trigger and diagnosis logic (B5).
Trigger: loss >= 3%  OR  loss >= 2% with (score >= 0.5 OR ensemble_std >= 0.3).
Diagnosis: rule-based classification into 10 categories.
"""

_VALID_DIAGNOSES = frozenset({
    "low_confidence_passed", "ensemble_divergence_ignored", "regime_mismatch",
    "news_staleness", "market_gap", "stop_too_tight",
    "correlated_portfolio_loss", "model_drift_active",
    "threshold_boundary", "unknown",
})

_CONFIDENCE_MIN   = 0.4
_STD_HIGH         = 0.30
_SCORE_BOUNDARY   = 0.30   # "threshold_boundary" if score near entry threshold
_SCORE_HIGH_CONV  = 0.50
_NEWS_MAX_AGE_MIN = 30.0


def should_trigger_postmortem(loss_pct: float, score: float, ensemble_std: float) -> bool:
    if loss_pct >= 0.03:
        return True
    if loss_pct >= 0.02 and (abs(score) >= _SCORE_HIGH_CONV or ensemble_std >= _STD_HIGH):
        return True
    return False


def diagnose_postmortem(
    loss_pct: float,
    score: float,
    confidence: float,
    ensemble_std: float,
    regime: str,
    reasoning: str,
    drift_active: bool,
    news_age_min: float,
) -> str:
    """
    Returns a diagnosis string from _VALID_DIAGNOSES.
    Rules are evaluated in priority order; first match wins.
    """
    if drift_active:
        return "model_drift_active"

    if news_age_min > _NEWS_MAX_AGE_MIN:
        return "news_staleness"

    if confidence < _CONFIDENCE_MIN:
        return "low_confidence_passed"

    if ensemble_std >= _STD_HIGH:
        return "ensemble_divergence_ignored"

    # Score within 0.03 of entry threshold → threshold_boundary
    if abs(abs(score) - _SCORE_BOUNDARY) < 0.03:
        return "threshold_boundary"

    if regime in ("risk_off", "high_vol") and score > 0:
        return "regime_mismatch"

    return "unknown"

# src/performance/threshold.py
"""
Threshold Optimizer: computes IC per score bucket and suggests a new entry
threshold if a tighter bucket has IC > 15% higher than the current threshold's bucket.
Never auto-applies — returns a suggestion only.
"""
from src.performance.ic import composite_ic

_BUCKETS: list[tuple[float, float]] = [
    (0.1, 0.2), (0.2, 0.3), (0.3, 0.4), (0.4, 0.6), (0.6, 1.0),
]
_CURRENT_THRESHOLD = 0.30
_IMPROVEMENT_THRESHOLD = 0.15   # 15% relative improvement required


def bucket_ic_analysis(
    scores: list[float],
    returns: list[float],
    confidences: list[float],
    current_threshold: float = _CURRENT_THRESHOLD,
) -> tuple[dict[str, float], float | None]:
    """
    Returns:
      bucket_ic: dict mapping "lo-hi" → composite_ic for that bucket
      suggestion: suggested new threshold if improvement > 15%, else None
    """
    bucket_ic: dict[str, float] = {}

    for lo, hi in _BUCKETS:
        indices = [
            i for i, s in enumerate(scores)
            if lo <= abs(s) < hi
        ]
        if len(indices) < 20:   # too few samples for this bucket
            continue
        s_b = [scores[i] for i in indices]
        r_b = [returns[i] for i in indices]
        c_b = [confidences[i] for i in indices]
        ic = composite_ic(s_b, r_b, c_b)
        bucket_ic[f"{lo:.1f}-{hi:.1f}"] = ic

    # Find IC at current threshold bucket
    current_key = None
    for lo, hi in _BUCKETS:
        if lo <= current_threshold < hi:
            current_key = f"{lo:.1f}-{hi:.1f}"
            break

    if current_key is None or current_key not in bucket_ic:
        return bucket_ic, None

    current_ic = bucket_ic[current_key]
    suggestion = None

    # Check if a stricter bucket (higher lo) has > 15% better IC
    for lo, hi in _BUCKETS:
        if lo <= current_threshold:
            continue  # only look at stricter thresholds
        key = f"{lo:.1f}-{hi:.1f}"
        if key not in bucket_ic:
            continue
        candidate_ic = bucket_ic[key]
        if current_ic > 0 and candidate_ic > current_ic * (1.0 + _IMPROVEMENT_THRESHOLD):
            suggestion = lo
            break

    return bucket_ic, suggestion
```

        forward_returns=all_returns,
        confidences=[0.75] * len(all_returns),
        current_weights=current_weights,
    )

    ctx = MarketContext(
        vix=float(redis.get("market:vix") or 20.0),
        vix_1d_change=0.0,
        portfolio_drawdown=float(redis.get("portfolio:drawdown") or 0.0),

        cross_asset_correlation=float(redis.get("market:cross_corr") or 0.5),
    )
    freeze, reason = should_freeze_weight_update(ctx)
    new_weights = compute_new_weights(purified, current_weights)

    suggestion_json = json.dumps({"suggested_weights": new_weights, "freeze_reason": reason})
    redis.set("ensemble:weights:suggestion", suggestion_json)

    msg = f"💡 Peso ensemble aggiornato (suggerimento, non applicato):\n"
    for m, w in new_weights.items():
        old_w = current_weights.get(m, 0)
        delta = w - old_w
        msg += f"  {m}: {old_w:.0%} → {w:.0%} ({delta:+.1%})\n"
    if freeze:
        msg += f"\n⛔ Circuit breaker attivo: {reason} — non applicato automaticamente."
    msg += "\nInvia /api/weights/approve per applicare."

def add_forward_returns(pg_conn, symbol: str, window_hours: int = 4):
    """Backfill forward returns into sentiment_signals using yfinance."""
    ticker = yf.Ticker(symbol)
    hist = ticker.history(period="3y", interval="1h")
    hist.index = hist.index.tz_localize("UTC") if hist.index.tzinfo is None else hist.index

    with pg_conn.cursor() as cur:
        cur.execute(
            "SELECT id, generated_at FROM sentiment_signals WHERE symbol = %s ORDER BY generated_at",
            (symbol,)
        )
        rows = cur.fetchall()

    updates = []
    for sig_id, gen_at in rows:
        future = gen_at + __import__("datetime").timedelta(hours=window_hours)
        try:
            price_now = float(hist.loc[hist.index >= gen_at, "Close"].iloc[0])
            price_future = float(hist.loc[hist.index >= future, "Close"].iloc[0])
            fwd_return = (price_future - price_now) / price_now
            updates.append((fwd_return, sig_id))
        except IndexError:
            continue

    with pg_conn.cursor() as cur:
        cur.executemany(
            "UPDATE sentiment_signals SET forward_return = %s WHERE id = %s",
            updates,
        )
    pg_conn.commit()
    print(f"  {symbol}: {len(updates)} forward returns added")
